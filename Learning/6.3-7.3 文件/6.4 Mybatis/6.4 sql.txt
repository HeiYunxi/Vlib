CREATE TABLE `users` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `username` varchar(255) DEFAULT NULL,
  `birthday` date DEFAULT NULL,
   `sex` varchar(2) DEFAULT NULL,
  `address` varchar(255) DEFAULT NULL,
 PRIMARY KEY (`id`)
);
